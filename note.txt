Các bước chạy app với quyền Administrator (Windows / PowerShell)
# 1. Mở PowerShell as admin
# 2. Vào thư mục root project
Set-Location -Path "D:\IT\ATBM-WebApplicationFirewall"

# 3. Tạo venv (nếu cần)
python -m venv venv

# 4. Kích hoạt venv (khi ở root)
.\venv\Scripts\activate

# nếu ở WAF và venv ở root
# ..\venv\Scripts\activate

# 5. Cài phụ thuộc
pip install -r requirements.txt
# hoặc
pip install flask numpy joblib scikit-learn

# 6. Chạy (từ gốc, hoặc cd vào WAF)
# Từ gốc, chạy module:
python -m WAF.FlaskSimpleTest

# Hoặc cd vào WAF rồi chạy file (sau khi activated venv)
cd WAF
python FlaskSimpleTest.py
